# SAGE OS Raspberry Pi Configuration
# This file should be placed in the boot partition of the SD card

# Set the kernel filename

# Disable rainbow splash screen

# Disable overscan
disable_overscan=1

# Set UART clock rate

# Enable UART for console
enable_uart=1
